Calling the API

The server runs locally at port 8500. All endpoints return JSON. No auth required for prototype builds.

// JavaScript — fetch platform stats
const response = await fetch('http://localhost:8500/api/db-stats');
const data = await response.json();
console.log(data.industries, data.blueprints);
// JavaScript — trigger an agent dry-run
const res = await fetch('http://localhost:8500/api/run/cg-replenishment-pr-agent', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ dry_run: true })
});
const result = await res.json();
# Python — fetch pending approvals
import requests
resp = requests.get("http://localhost:8500/api/approvals")
approvals = resp.json()
for item in approvals:
    print(item["item_id"], item["decision"])
